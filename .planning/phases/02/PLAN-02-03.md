## Test Categories

### Category 1: WAL Corruption Scenarios

1. **Truncated WAL file** - WAL ends mid-record
2. **Corrupted record header** - Invalid magic bytes or version
3. **Corrupted record payload** - Valid header, invalid data
4. **Checksum mismatch** - Record checksum doesn't match

### Category 2: Transaction Edge Cases

5. **Incomplete transaction (no commit)** - Transaction begins but never commits
6. **Rollback after partial writes** - Transaction with some records, then rollback
7. **Duplicate transaction IDs** - Two transactions with same ID
8. **Commit without begin** - Commit record without matching begin

### Category 3: Checkpoint Edge Cases

9. **Incomplete checkpoint** - Checkpoint interrupted mid-operation
10. **Stale checkpoint** - Checkpoint LSN older than some committed transactions
11. **Checkpoint with corruption** - Checkpoint data is corrupted
12. **Missing checkpoint file** - Checkpoint referenced but file missing

### Category 4: Recovery Scenarios

13. **Empty WAL file** - WAL exists but has no records
14. **WAL with only committed transactions** - Normal case
15. **WAL with mix of committed and uncommitted** - Mixed state
16. **Recovery after crash during checkpoint** - WAL and checkpoint inconsistent

```rust
//! WAL Recovery Edge Case Tests

use sqlitegraph::backend::native::v2::wal::{
    V2WALManager, V2WALConfig, V2WALRecord,
};
use tempfile::TempDir;

struct RecoveryTestSetup {
    temp_dir: TempDir,
    wal_path: PathBuf,
    db_path: PathBuf,
}

impl RecoveryTestSetup {
    fn new() -> Self {
        let temp_dir = TempDir::new().unwrap();
        let wal_path = temp_dir.path().join("test.wal");
        let db_path = temp_dir.path().join("test.db");
        Self { temp_dir, wal_path, db_path }
    }

    fn write_corrupted_wal(&self, data: &[u8]) {
        std::fs::write(&self.wal_path, data).unwrap();
    }

    fn write_truncated_wal(&self, records: Vec<V2WALRecord>, truncate_at: usize) {
        // Write records, then truncate at byte offset
        let mut data = Vec::new();
        for record in &records {
            let serialized = bincode::serialize(record).unwrap();
            data.extend_from_slice(&serialized);
        }
        data.truncate(truncate_at);
        std::fs::write(&self.wal_path, data).unwrap();
    }
}
```

```rust
#[test]
fn test_recovery_from_truncated_wal() {
    let setup = RecoveryTestSetup::new();

    // Create a WAL with some records, then truncate mid-record
    let records = vec![
        V2WALRecord::TransactionBegin { tx_id: 1 },
        V2WALRecord::NodeInsert { node_id: 1, data: vec![1, 2, 3] },
        V2WALRecord::TransactionCommit { tx_id: 1, lsn: 100 },
    ];

    // Truncate at 50 bytes (mid-second-record)
    setup.write_truncated_wal(records, 50);

    // Recovery should:
    // 1. Detect corruption
    // 2. Recover complete transaction before truncation
    // 3. Warn about truncated data

    let config = V2WALConfig {
        wal_path: setup.wal_path.clone(),
        force_recovery: true,
        ..Default::default()
    };

    let result = V2WALManager::recover(&config);
    assert!(result.is_ok());

    let recovery_result = result.unwrap();
    assert!(recovery_result.warnings.len() > 0);
    assert!(recovery_result.warnings.iter().any(|w| w.contains("truncated")));
}
```

### Task 02-03d: Test Corrupted Checksum

```rust
#[test]
fn test_recovery_with_corrupted_checksum() {
    let setup = RecoveryTestSetup::new();

    // Write WAL and corrupt a record checksum
    let manager = V2WALManager::new(setup.wal_path.clone()).unwrap();
    manager.begin_transaction(1).unwrap();
    manager.write_record(V2WALRecord::NodeInsert { node_id: 1, data: vec![] }).unwrap();
    manager.commit_transaction(1).unwrap();

    // Corrupt the WAL file
    let mut wal_data = std::fs::read(&setup.wal_path).unwrap();
    if wal_data.len() > 50 {
        wal_data[50] = wal_data[50].wrapping_add(1); // Flip a byte
    }
    std::fs::write(&setup.wal_path, wal_data).unwrap();

    // Recover with corruption detection
    let config = V2WALConfig {
        wal_path: setup.wal_path.clone(),
        force_recovery: false, // Don't force - should fail or warn
        ..Default::default()
    };

    let result = V2WALManager::recover(&config);

    // Should either fail or succeed with warnings
    match result {
        Ok(recovery_result) => {
            assert!(recovery_result.warnings.iter().any(|w| w.contains("corruption") || w.contains("checksum")));
        }
        Err(e) => {
            assert!(e.to_string().contains("corruption") || e.to_string().contains("checksum"));
        }
    }
}
```

```rust
#[test]
fn test_recovery_after_crash_during_checkpoint() {
    let setup = RecoveryTestSetup::new();

    // Setup: Create WAL and checkpoint
    let manager = V2WALManager::new(setup.wal_path.clone()).unwrap();

    // Write some transactions
    for i in 1..=10 {
        manager.begin_transaction(i).unwrap();
        manager.write_record(V2WALRecord::NodeInsert { node_id: i, data: vec![] }).unwrap();
        manager.commit_transaction(i).unwrap();
    }

    // Start checkpoint (simulate mid-checkpoint crash)
    let checkpoint_result = manager.checkpoint();

    // Simulate crash: don't wait for checkpoint to complete
    // Instead, immediately try to recover

    drop(manager); // Drop to simulate crash

    // Recover
    let config = V2WALConfig {
        wal_path: setup.wal_path.clone(),
        ..Default::default()
    };

    let result = V2WALManager::recover(&config);
    assert!(result.is_ok());

    let recovery_result = result.unwrap();
    // Should recover all 10 committed transactions
    assert_eq!(recovery_result.committed_transactions, 10);
}
```

```rust
#[test]
fn test_checkpoint_corruption_detection() {
    let setup = RecoveryTestSetup::new();

    // Create a checkpoint file
    let checkpoint_path = setup.temp_dir.path().join("checkpoint.chk");
    std::fs::write(&checkpoint_path, vec![0u8; 1024]).unwrap();

    // Corrupt the checkpoint
    let mut checkpoint_data = std::fs::read(&checkpoint_path).unwrap();
    checkpoint_data[0] = 0xFF; // Corrupt header
    std::fs::write(&checkpoint_path, checkpoint_data).unwrap();

    // Try to recover with corrupted checkpoint
    let config = V2WALConfig {
        wal_path: setup.wal_path.clone(),
        checkpoint_path: Some(checkpoint_path),
        ..Default::default()
    };

    let result = V2WALManager::recover(&config);

    // Should detect checkpoint corruption
    match result {
        Ok(recovery_result) => {
            assert!(recovery_result.warnings.iter().any(|w| w.contains("checkpoint") && w.contains("corruption")));
        }
        Err(e) => {
            assert!(e.to_string().contains("checkpoint") || e.to_string().contains("corruption"));
        }
    }
}
```
