```rust
fn run_hnsw_list(client: &BackendClient, args: &[String]) -> Result<(), SqliteGraphError> {
    let graph = client.graph().ok_or_else(|| SqliteGraphError::invalid_input("hnsw-list requires SQLite backend"))?;

    // Get list of index names from in-memory registry (loaded on startup)
    let index_names = graph.list_hnsw_indexes()?;

    // Also query database directly for persisted metadata
    let conn = &graph.conn;
    let mut indexes_info = Vec::new();

    for name in &index_names {
        // Get detailed info for each index
        if let Ok(Some(info)) = get_index_info(conn, name) {
            indexes_info.push(info);
        }
    }

    let payload = json!({
        "command": "hnsw-list",
        "count": index_names.len(),
        "indexes": indexes_info,
        "status": "completed"
    });
    println!("{payload}");
    Ok(())
}

// Helper to get index metadata from database
fn get_index_info(conn: &Connection, name: &str) -> Result<Option<serde_json::Value>, SqliteGraphError> {
    conn.query_row(
        "SELECT dimension, m, ef_construction, distance_metric, vector_count FROM hnsw_indexes WHERE name = ?",
        &[name],
        |row| {
            Ok(json!({
                "name": name,
                "dimension": row.get::<_, i64>(0)?,
                "m": row.get::<_, i64>(1)?,
                "ef_construction": row.get::<_, i64>(2)?,
                "distance_metric": row.get::<_, String>(3)?,
                "vector_count": row.get::<_, i64>(4)?
            }))
        }
    ).optional().map_err(|e| SqliteGraphError::invalid_input(format!("Failed to query index info: {}", e)))
}
```

```rust
fn run_hnsw_delete(client: &BackendClient, args: &[String]) -> Result<(), SqliteGraphError> {
    let graph = client.graph().ok_or_else(|| SqliteGraphError::invalid_input("hnsw-delete requires SQLite backend"))?;

    // Get index name from --index-name or --name parameter
    let index_name = args.iter()
        .position(|arg| arg == "--index-name" || arg == "--name")
        .and_then(|idx| args.get(idx + 1))
        .map(|s| s.as_str())
        .ok_or_else(|| SqliteGraphError::invalid_input("--index-name is required for hnsw-delete"))?;

    // Check if index exists
    let exists = graph.list_hnsw_indexes()?.iter().any(|n| n == index_name);
    if !exists {
        let payload = json!({
            "command": "hnsw-delete",
            "index_name": index_name,
            "error": "Index not found",
            "status": "error"
        });
        println!("{payload}");
        return Ok(()); // Don't error out, just report
    }

    // Delete from database (CASCADE handles vectors)
    let deleted = HnswIndex::delete_index(&graph.conn, index_name)
        .map_err(|e| SqliteGraphError::invalid_input(format!("Failed to delete index: {}", e)))?;

    // Remove from in-memory registry
    {
        use std::sync::RwLock;
        let mut indexes = graph.hnsw_indexes.write()
            .map_err(|e| SqliteGraphError::invalid_input(format!("RwLock poisoned: {}", e)))?;
        indexes.remove(index_name);
    }

    let payload = json!({
        "command": "hnsw-delete",
        "index_name": index_name,
        "deleted": deleted,
        "status": "completed"
    });
    println!("{payload}");
    Ok(())
}
```

```rust
fn run_hnsw_info(client: &BackendClient, args: &[String]) -> Result<(), SqliteGraphError> {
    let graph = client.graph().ok_or_else(|| SqliteGraphError::invalid_input("hnsw-info requires SQLite backend"))?;

    // Get index name (default to "default" if not specified)
    let index_name = args.iter()
        .position(|arg| arg == "--index-name" || arg == "--name")
        .and_then(|idx| args.get(idx + 1))
        .map(|s| s.as_str())
        .unwrap_or("default");

    // Check if index exists
    let exists = graph.list_hnsw_indexes()?.iter().any(|n| n == index_name);
    if !exists {
        let payload = json!({
            "command": "hnsw-info",
            "index_name": index_name,
            "error": "Index not found",
            "status": "error"
        });
        println!("{payload}");
        return Ok(());
    }

    // Get metadata from database
    let metadata = get_index_info(&graph.conn, index_name)?;

    // Get detailed statistics from the index
    let stats_result = graph.get_hnsw_index_ref(index_name, |hnsw| {
        hnsw.statistics()
    });

    let payload = match stats_result {
        Ok(Ok(stats)) => json!({
            "command": "hnsw-info",
            "index_name": index_name,
            "metadata": metadata,
            "statistics": {
                "vector_count": stats.vector_count,
                "layer_count": stats.layer_count,
                "entry_point_count": stats.entry_point_count,
                "dimension": stats.dimension,
                "distance_metric": format!("{:?}", stats.distance_metric),
                "storage": {
                    "backend_type": stats.storage_stats.backend_type,
                    "estimated_memory_bytes": stats.storage_stats.estimated_memory_bytes
                },
                "layers": stats.layer_stats.iter()
                    .map(|(layer, nodes, conn)| json!({
                        "layer": layer,
                        "node_count": nodes,
                        "avg_connections": conn
                    }))
                    .collect::<Vec<_>>()
            },
            "status": "completed"
        }),
        Ok(Err(e)) => json!({
            "command": "hnsw-info",
            "index_name": index_name,
            "metadata": metadata,
            "error": e.to_string(),
            "status": "partial"
        }),
        Err(e) => json!({
            "command": "hnsw-info",
            "index_name": index_name,
            "metadata": metadata,
            "error": e.to_string(),
            "status": "partial"
        })
    };

    println!("{payload}");
    Ok(())
}
```
