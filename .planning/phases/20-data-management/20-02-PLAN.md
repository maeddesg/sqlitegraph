    Pattern:
    ```rust
    use crate::backend::native::{
        constants::MAGIC_BYTES,
        types::{NativeBackendError, NativeResult},
    };

    pub fn detect_format_version(path: &Path) -> NativeResult<FormatVersion> {
        use std::fs::File;
        use std::io::Read;

        let mut file = File::open(path)
            .map_err(|e| NativeBackendError::Io(e))?;

        let mut header = [0u8; 80];
        file.read_exact(&mut header)
            .map_err(|e| NativeBackendError::Io(e))?;

        // Verify magic
        let magic = &header[0..8];
        if magic != MAGIC_BYTES {
            return Err(NativeBackendError::InvalidMagic { ... });
        }

        // Read version (offset 8-11)
        let version_bytes = [header[8], header[9], header[10], header[11]];
        let version = u32::from_be_bytes(version_bytes);

        Ok(match version {
            1 => FormatVersion::V1,
            2 => FormatVersion::V2,
            3 => FormatVersion::V3,
            v => FormatVersion::Unknown(v),
        })
    }
    ```

    2. migrate_file function with atomic rollback pattern:
       - Create backup at path.with_extension("bak")
       - Use AtomicFileOperations::atomic_copy_file for backup
       - Read entire file content
       - For v2->v3 migration: Convert header in memory
       - Write to temp file with .tmp extension
       - Sync temp file to disk
       - Atomic rename temp -> original
       - Sync parent directory
       - Verify migration by reopening file
       - Delete backup on success
       - Rollback: On any error, restore backup using atomic copy

    Pattern:
    ```rust
    use crate::backend::native::{
        v2::snapshot::AtomicFileOperations,
        graph_file::encoding::{decode_persistent_header, encode_persistent_header},
        persistent_header::PersistentHeaderV2,
        types::{NativeBackendError, NativeResult},
    };
    use std::path::Path;
    use std::time::Instant;

    pub fn migrate_file(path: &Path) -> NativeResult<MigrationResult> {
        let start = Instant::now();
        let backup_path = path.with_extension("bak");

        // Step 1: Create backup
        AtomicFileOperations::new()
            .atomic_copy_file(path, &backup_path)
            .map_err(|e| MigrationError::BackupFailed { source: format!("{:?}", e) })?;

        // Step 2: Read and convert header
        let content = std::fs::read(path)?;
        let header = decode_persistent_header(&content)?;

        // Step 3: Convert v2 to v3 (if needed)
        let migrated_content = if header.version == 2 {
            convert_v2_to_v3(&content, &header)?
        } else {
            content // Already at current version
        };

        // Step 4: Write to temp file
        let temp_path = path.with_extension("tmp");
        std::fs::write(&temp_path, migrated_content)?;
        sync_file(&temp_path)?;

        // Step 5: Atomic rename
        std::fs::rename(&temp_path, path)?;
        sync_directory(path.parent().unwrap())?;

        // Step 6: Verify
        let verify_header = decode_persistent_header(&std::fs::read(path)?)?;
        if verify_header.version != 3 {
            return rollback(backup_path, path, MigrationError::VerificationFailed {
                reason: format!("Version is {} not 3", verify_header.version)
            });
        }

        // Step 7: Cleanup backup
        let _ = std::fs::remove_file(&backup_path);

        Ok(MigrationResult {
            from_version: 2,
            to_version: 3,
            backup_path,
            duration: start.elapsed(),
        })
    }

    fn convert_v2_to_v3(content: &[u8], header: &PersistentHeaderV2) -> NativeResult<Vec<u8>> {
        // Update header version to 3
        let mut new_header = header.clone();
        new_header.version = 3;
        // schema_version and reserved are already correct (decode handled conversion)

        let mut new_content = content.to_vec();

        // Write updated header (first 80 bytes)
        let encoded = encode_persistent_header(&new_header)?;
        new_content[0..80].copy_from_slice(&encoded);

        Ok(new_content)
    }

    fn rollback(backup_path: &Path, original_path: &Path, error: MigrationError) -> NativeResult<!> {
        let _ = AtomicFileOperations::new()
            .atomic_copy_file(backup_path, original_path);
        let _ = std::fs::remove_file(&backup_path);
        Err(NativeBackendError::MigrationFailed { source: format!("{:?}", error) }.into())
    }
    ```

    3. Helper functions for sync operations (reuse from AtomicFileOperations)
