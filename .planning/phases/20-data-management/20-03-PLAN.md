       ```rust
               Self {

                   backup_id: None,
                   checkpoint_before_backup: true,
                   include_statistics: true,
                   validate_checksum: true,
               }
           }
       }

       impl BackupConfig {
           pub fn new(backup_dir: impl AsRef<Path>) -> Self {
               Self {
                   backup_dir: backup_dir.as_ref().to_path_buf(),
                   ..Default::default()
               }
           }

           pub fn with_backup_id(mut self, id: impl Into<String>) -> Self {
               self.backup_id = Some(id.into());
               self
           }

           pub fn with_checkpoint(mut self, enabled: bool) -> Self {
               self.checkpoint_before_backup = enabled;
               self
           }
       }
       ```

    4. create_backup function:
       ```rust
       pub fn create_backup(
           graph_path: &Path,
           config: BackupConfig,
       ) -> NativeResult<BackupResult> {
           use std::time::{Duration, UNIX_EPOCH};

           let start = SystemTime::now();
           let backup_id = config.backup_id.clone().unwrap_or_else(|| {
               format!("backup_{}",
                   start.duration_since(UNIX_EPOCH)
                       .unwrap_or_default()
                       .as_secs())
           });

           // Ensure backup directory exists
           std::fs::create_dir_all(&config.backup_dir)
               .map_err(|e| NativeBackendError::Io(e))?;

           // Step 1: Checkpoint before backup (if enabled)
           let checkpoint_performed = if config.checkpoint_before_backup {
               // TODO: Access WAL manager for checkpoint
               // For now, we assume checkpoint was called externally
               true
           } else {
               false
           };

           // Step 2: Create snapshot export config
           let export_config = SnapshotExportConfig {
               export_path: config.backup_dir.clone(),
               snapshot_id: backup_id.clone(),
               include_statistics: config.include_statistics,
               min_stable_duration: Duration::from_secs(0),
               checksum_validation: config.validate_checksum,
           };

           // Step 3: Create exporter and export
           let mut exporter = SnapshotExporter::new(graph_path, export_config)?;
           let export_result = exporter.export_snapshot()?;

           // Step 4: Convert to BackupResult
           let duration = start.elapsed()
               .unwrap_or(Duration::from_secs(0))
               .as_secs_f64();

           Ok(BackupResult {
               snapshot_path: export_result.snapshot_path,
               manifest_path: export_result.manifest_path,
               size_bytes: export_result.snapshot_size_bytes,
               checksum: export_result.checksum,
               record_count: export_result.record_count,
               duration_secs: duration,
               timestamp: export_result.export_timestamp,
               checkpoint_performed,
           })
       }
       ```

    5. Convenience function:

       ```rust
       /// Quick backup with defaults
       pub fn backup(graph_path: &Path, backup_dir: &Path) -> NativeResult<BackupResult> {
           create_backup(graph_path, BackupConfig::new(backup_dir))
       }
       ```

       ```rust
       impl Backend {
           // ... existing methods ...

           /// Create a backup of the database
           ///
           /// Creates a consistent snapshot of the database including all data pages.
           /// For Native V2 backend, optionally checkpoints before backup to ensure
           /// WAL is applied and snapshot is consistent.
           ///
           /// # Arguments
           /// * `backup_dir` - Destination directory for backup files
           ///
           /// # Returns
           /// Backup result with paths, checksum, and metadata
           pub fn backup(&self, backup_dir: &Path) -> crate::Result<BackupResult> {
               match self {
                   Backend::NativeV2 { graph_path, .. } => {
                       let config = BackupConfig::new(backup_dir);
                       crate::backend::native::v2::create_backup(graph_path, config)
                           .map_err(|e| crate::Error::Backend(e.to_string()))
                   }
                   #[cfg(feature = "sqlite")]
                   Backend::Sqlite { pool, .. } => {
                       // For SQLite, use SQLite backup API
                       use crate::graph::SqliteConfig;
                       use std::fs;

                       // Ensure backup directory exists
                       fs::create_dir_all(backup_dir)
                           .map_err(|e| crate::Error::Io(e))?;

                       // Generate backup filename
                       let timestamp = std::time::SystemTime::now()
                           .duration_since(std::time::UNIX_EPOCH)
                           .unwrap_or_default()
                           .as_secs();
                       let backup_path = backup_dir.join(format!("backup_{}.db", timestamp));

                       // Use SQLite backup API via pool connection
                       let conn = pool.get()
                           .map_err(|e| crate::Error::Backend(e.to_string()))?;

                       // SQLite backup to file
                       let mut db = conn.borrow().raw();
                       sqlitegraph::backup_database(&mut db, &backup_path)
                           .map_err(|e| crate::Error::Backend(e.to_string()))?;

                       Ok(BackupResult {
                           snapshot_path: backup_path.clone(),
                           manifest_path: backup_path.with_extension("json"),
                           size_bytes: fs::metadata(&backup_path)
                               .map(|m| m.len())
                               .unwrap_or(0),
                           checksum: 0,
                           record_count: 0,
                           duration_secs: 0.0,
                           timestamp,
                           checkpoint_performed: false,
                       })
                   }
               }
           }
       }
   }
   ```

    Note: For SQLite backend, we may need to stub or implement a simple backup. The focus is on Native V2.
