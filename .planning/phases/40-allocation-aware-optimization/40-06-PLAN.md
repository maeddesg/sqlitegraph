    fn test_crash_replay_rebuilds_delta() {
        let mut backend1 = create_test_backend();
        let graph_id = backend1.create_graph("test").unwrap();

        // Perform operations
        backend1.add_node(graph_id, 1, &[], &[]).unwrap();
        backend1.checkpoint().unwrap();

        let tx1 = backend1.begin_transaction().unwrap();
        backend1.add_node_in_tx(graph_id, 2, &[], &[], tx1).unwrap();
        backend1.commit_transaction(tx1).unwrap();

        let tx2 = backend1.begin_transaction().unwrap();
        backend1.add_node_in_tx(graph_id, 3, &[], &[], tx2).unwrap();
        backend1.commit_transaction(tx2).unwrap();

        // Simulate crash and recovery (WAL replay rebuilds delta index)
        let wal_data = backend1.get_wal_data();
        let mut backend2 = create_test_backend();
        backend2.recover_from_wal(wal_data).unwrap();

        // Both backends should have identical state
        let snapshot = SnapshotId::current();

        let nodes1 = backend1.iter_nodes(graph_id, snapshot).unwrap()
            .collect::<Vec<_>>();
        let nodes2 = backend2.iter_nodes(graph_id, snapshot).unwrap()
            .collect::<Vec<_>>();

        assert_eq!(nodes1, nodes2);
    }

    #[test]
    fn test_wal_replay_snapshot_isolation_with_delta() {
        let mut backend = create_test_backend();
        let graph_id = backend.create_graph("test").unwrap();

        // Checkpoint
        backend.add_node(graph_id, 1, &[], &[]).unwrap();
        backend.checkpoint().unwrap();

        // Transaction 1
        let tx1 = backend.begin_transaction().unwrap();
        backend.add_node_in_tx(graph_id, 2, &[], &[], tx1).unwrap();
        let commit_lsn_1 = backend.commit_transaction(tx1).unwrap();

        // Transaction 2
        let tx2 = backend.begin_transaction().unwrap();
        backend.add_node_in_tx(graph_id, 3, &[], &[], tx2).unwrap();
        backend.commit_transaction(tx2).unwrap();

        // Crash and recover (WAL replay rebuilds delta index)
        let wal_data = backend.get_wal_data();
        let mut recovered = create_test_backend();
        recovered.recover_from_wal(wal_data).unwrap();

        assert!(!recovered.has_node(graph_id, 3, snapshot).unwrap()); // tx2 not visible (future delta)
