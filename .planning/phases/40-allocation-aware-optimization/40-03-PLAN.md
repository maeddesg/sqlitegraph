# 40-03: Implement Committed-Delta Index (Architecture Revision)

<objective>
Implement an in-memory delta index that tracks committed changes at commit time, providing O(1) overlay lookup for snapshot isolation without WAL scanning.

**Purpose:** The original plan 40-03 assumed a WAL-overlay per-read architecture, but the actual codebase uses mmap-based direct file reads. This plan implements the correct approach: build a delta index at commit time, then read operations can quickly overlay committed-but-not-checkpointed changes via O(1) HashMap lookups.

**Output:** DeltaIndex module that tracks committed changes with LSN tags, enabling snapshot-aware reads without performance-killing WAL scanning.
</objective>

    //! Committed-Delta Index for Snapshot Isolation
    //!
    //! This module implements an in-memory delta map that tracks committed changes
    //! at transaction commit time, enabling O(1) overlay lookup for snapshot-aware
    //! reads without WAL scanning.
    //!
    //! # Architecture
    //!
    //! - DeltaIndex stores committed changes as HashMap<entity_id, Vec<(value, commit_lsn)>>
    //! - At commit time: transaction changes are appended with their commit_lsn
    //! - At read time: check delta for entity, apply if commit_lsn <= snapshot_id
    //! - At checkpoint time: drop deltas that are now in the checkpointed base
    //!
    //! # Why This Approach
    //!
    //! The native backend uses mmap-based direct file reads (~100ns), not WAL overlay.
    //! Scanning WAL on every read would be 1000-10000x slower. The delta index provides
    //! O(1) lookups for recent committed-but-not-checkpointed changes, preserving the
    //! fast mmap path for checkpointed data.

    use std::collections::HashMap;
    use std::sync::{Arc, RwLock};

    use crate::types::{NodeId, EdgeId};
    use crate::snapshot::SnapshotId;
    use crate::backend::native::v2::wal::WalRecord;

    /// Delta record tracking a committed change
    #[derive(Debug, Clone)]
    pub struct DeltaRecord {
        /// The committed WAL record
        pub record: WalRecord,
        /// LSN when this record was committed
        pub commit_lsn: u64,
    }

    /// In-memory delta index for committed-but-not-checkpointed changes
    ///
    /// Tracks node and edge changes committed but not yet checkpointed,
    /// enabling snapshot-aware reads via O(1) HashMap lookup instead of
    /// WAL scanning.
    #[derive(Debug, Clone)]
    pub struct DeltaIndex {
        /// Node deltas: node_id -> Vec<(record, commit_lsn)>
        node_deltas: HashMap<NodeId, Vec<DeltaRecord>>,
        /// Edge deltas: edge_id -> Vec<(record, commit_lsn)>
        edge_deltas: HashMap<EdgeId, Vec<DeltaRecord>>,
    }

    impl Default for DeltaIndex {
        fn default() -> Self {
            Self::new()
        }
    }

    impl DeltaIndex {
        /// Create a new empty delta index
        pub fn new() -> Self {
            Self {
                node_deltas: HashMap::new(),
                edge_deltas: HashMap::new(),
            }
        }

        /// Apply committed transaction changes to delta index
        ///
        /// Called at transaction commit time with:
        /// - The WAL records written by this transaction
        /// - The commit_lsn assigned to this transaction
        ///
        /// This builds the delta map incrementally - each commit appends
        /// its changes to the index, tagged with the commit_lsn.
        pub fn apply_commit(
            &mut self,
            records: Vec<WalRecord>,
            commit_lsn: u64,
        ) {
            for record in records {
                match &record {
                    WalRecord::AddNode { node_id, .. } |
                    WalRecord::DeleteNode { node_id, .. } => {
                        let deltas = self.node_deltas.entry(*node_id).or_insert_with(Vec::new);
                        deltas.push(DeltaRecord {
                            record,
                            commit_lsn,
                        });
                    }

                    WalRecord::AddEdge { edge_id, .. } |
                    WalRecord::DeleteEdge { edge_id, .. } => {
                        let deltas = self.edge_deltas.entry(*edge_id).or_insert_with(Vec::new);
                        deltas.push(DeltaRecord {
                            record,
                            commit_lsn,
                        });
                    }

                    // Records that don't need delta tracking
                    WalRecord::BeginTransaction { .. } |
                    WalRecord::CommitTransaction { .. } |
                    WalRecord::RollbackTransaction { .. } => {
                        // Transaction metadata, no entity delta
                    }

                    // TODO: Handle other record types as needed
                    _ => {}
                }
            }
        }

        /// Get the most recent visible delta for a node
        ///
        /// Returns the most recent delta record where commit_lsn <= snapshot_id.
        /// Returns None if no visible delta exists (either no deltas, or all
        /// deltas are from transactions after the snapshot).
        pub fn get_node_delta(
            &self,
            node_id: NodeId,
            snapshot_id: SnapshotId,
        ) -> Option<&DeltaRecord> {
            let snapshot_lsn = snapshot_id.as_lsn();

            self.node_deltas.get(&node_id)
                .and_then(|deltas| {
                    // Find most recent delta where commit_lsn <= snapshot_lsn
                    // Deltas are in commit order (ascending), so search from end
                    deltas.iter()
                        .rev()
                        .find(|delta| delta.commit_lsn <= snapshot_lsn)
                })
        }

        /// Get the most recent visible delta for an edge
        pub fn get_edge_delta(
            &self,
            edge_id: EdgeId,
            snapshot_id: SnapshotId,
        ) -> Option<&DeltaRecord> {
            let snapshot_lsn = snapshot_id.as_lsn();

            self.edge_deltas.get(&edge_id)
                .and_then(|deltas| {
                    deltas.iter()
                        .rev()
                        .find(|delta| delta.commit_lsn <= snapshot_lsn)
                })
        }

        /// Remove all deltas up to and including the given checkpoint LSN
        ///
        /// Called after checkpoint completes. All changes with commit_lsn <= checkpoint_lsn
        /// are now in the checkpointed base, so we can drop them from the delta index.
        pub fn checkpoint_completed(&mut self, checkpoint_lsn: u64) {
            // Remove node deltas checkpointed
            self.node_deltas.retain(|_node_id, deltas| {
                // Keep only deltas after checkpoint_lsn
                deltas.retain(|delta| delta.commit_lsn > checkpoint_lsn);
                // Keep entry if there are still uncheckpointed deltas
                !deltas.is_empty()
            });

            // Remove edge deltas checkpointed
            self.edge_deltas.retain(|_edge_id, deltas| {
                deltas.retain(|delta| delta.commit_lsn > checkpoint_lsn);
                !deltas.is_empty()
            });
        }

        /// Clear all deltas (for testing or rollback scenarios)
        #[cfg(test)]
        pub fn clear(&mut self) {
            self.node_deltas.clear();
            self.edge_deltas.clear();
        }

        /// Get the number of node deltas (for testing/metrics)
        #[cfg(test)]
        pub fn node_delta_count(&self) -> usize {
            self.node_deltas.len()
        }

        /// Get the number of edge deltas (for testing/metrics)
        #[cfg(test)]
        pub fn edge_delta_count(&self) -> usize {
            self.edge_deltas.len()
        }
    }

    /// Thread-safe wrapper for DeltaIndex
    ///
    /// Allows shared access across the backend while ensuring
    /// thread-safe reads and writes to the delta map.
    #[derive(Debug, Clone)]
    pub struct SharedDeltaIndex {
        inner: Arc<RwLock<DeltaIndex>>,
    }

    impl SharedDeltaIndex {
        /// Create a new shared delta index
        pub fn new() -> Self {
            Self {
                inner: Arc::new(RwLock::new(DeltaIndex::new())),
            }
        }

        /// Apply committed transaction changes
        pub fn apply_commit(
            &self,
            records: Vec<WalRecord>,
            commit_lsn: u64,
        ) {
            let mut index = self.inner.write()
                .expect("DeltaIndex lock poisoned");
            index.apply_commit(records, commit_lsn);
        }

        /// Get the most recent visible delta for a node
        pub fn get_node_delta(
            &self,
            node_id: NodeId,
            snapshot_id: SnapshotId,
        ) -> Option<DeltaRecord> {
            let index = self.inner.read()
                .expect("DeltaIndex lock poisoned");
            index.get_node_delta(node_id, snapshot_id).cloned()
        }

        /// Get the most recent visible delta for an edge
        pub fn get_edge_delta(
            &self,
            edge_id: EdgeId,
            snapshot_id: SnapshotId,
        ) -> Option<DeltaRecord> {
            let index = self.inner.read()
                .expect("DeltaIndex lock poisoned");
            index.get_edge_delta(edge_id, snapshot_id).cloned()
        }

        /// Remove all deltas up to and including checkpoint LSN
        pub fn checkpoint_completed(&self, checkpoint_lsn: u64) {
            let mut index = self.inner.write()
                .expect("DeltaIndex lock poisoned");
            index.checkpoint_completed(checkpoint_lsn);
        }
    }

    impl Default for SharedDeltaIndex {
        fn default() -> Self {
            Self::new()
        }
    }

            assert_eq!(index.edge_delta_count(), 0);
        }

            let mut index = DeltaIndex::new();

                    graph_id: GraphId::new(1),
                    node_id: NodeId::new(1),
                    labels: vec![],
                    properties: vec![],
                },
                WalRecord::CommitTransaction { tx_id: 1, commit_lsn: 100 },
            ];

            index.apply_commit(records, 100);

            assert_eq!(index.node_delta_count(), 1);
        }

        #[test]
        fn test_get_node_delta_visible() {
            let mut index = DeltaIndex::new();

            // Commit at LSN 100

            ];
            index.apply_commit(records, 100);

            let snapshot = SnapshotId::from_lsn(150);

            assert!(delta.is_some());
            assert_eq!(delta.unwrap().commit_lsn, 100);
        }

        #[test]
        fn test_get_node_delta_not_visible() {
            let mut index = DeltaIndex::new();

            // Commit at LSN 200
            let records = vec![
                WalRecord::AddNode {
                    graph_id: GraphId::new(1),
                    node_id: NodeId::new(1),
                    labels: vec![],
                    properties: vec![],
                },
            ];
            index.apply_commit(records, 200);

            // Snapshot at LSN 150 should NOT see the delta
            let snapshot = SnapshotId::from_lsn(150);
            let delta = index.get_node_delta(NodeId::new(1), snapshot);

            assert!(delta.is_none());
        }

        #[test]
        fn test_checkpoint_completed() {
            let mut index = DeltaIndex::new();

            // Commit at LSN 100
            let records = vec![
                WalRecord::AddNode {
                    graph_id: GraphId::new(1),
                    node_id: NodeId::new(1),
                    labels: vec![],
                    properties: vec![],
                },
            ];
            index.apply_commit(records, 100);

            assert_eq!(index.node_delta_count(), 1);

            // Checkpoint at LSN 150 should drop the delta
            index.checkpoint_completed(150);

            assert_eq!(index.node_delta_count(), 0);
        }

        #[test]
        fn test_multiple_deltas_same_entity() {
            let mut index = DeltaIndex::new();

            // First commit: Add node
            let records1 = vec![
                WalRecord::AddNode {
                    graph_id: GraphId::new(1),
                    node_id: NodeId::new(1),
                    labels: vec![],
                    properties: vec![],
                },
            ];
            index.apply_commit(records1, 100);

            // Second commit: Delete node
            let records2 = vec![
                WalRecord::DeleteNode {
                    graph_id: GraphId::new(1),
                    node_id: NodeId::new(1),
                },
            ];
            index.apply_commit(records2, 200);

            // Snapshot at LSN 150 should see AddNode (not DeleteNode)
            let snapshot1 = SnapshotId::from_lsn(150);
            let delta1 = index.get_node_delta(NodeId::new(1), snapshot1);
            assert!(delta1.is_some());
            assert!(matches!(delta1.unwrap().record, WalRecord::AddNode { .. }));

            // Snapshot at LSN 250 should see DeleteNode (most recent)
            let snapshot2 = SnapshotId::from_lsn(250);
            let delta2 = index.get_node_delta(NodeId::new(1), snapshot2);
            assert!(delta2.is_some());
            assert!(matches!(delta2.unwrap().record, WalRecord::DeleteNode { .. }));
        }
    }
