    ```rust
    #[cfg(test)]
    mod integration_tests {
        use super::*;
        use crate::types::GraphId;

        #[test]
        fn test_delta_index_lifecycle() {
            // Setup: Create WAL manager with delta index
            let manager = V2WALManager::new_test().unwrap();

            // Commit 1: Add node
            let tx1 = manager.begin_transaction().unwrap();
            manager.add_node_in_tx(
                GraphId::new(1),
                NodeId::new(1),
                vec![],
                vec![],
                tx1,
            ).unwrap();
            let commit_lsn1 = manager.commit_transaction(tx1).unwrap();

            // Verify delta index populated
            let delta = manager.delta_index()
                .get_node_delta(NodeId::new(1), SnapshotId::from_lsn(commit_lsn1));
            assert!(delta.is_some());
            assert_eq!(delta.unwrap().commit_lsn, commit_lsn1);

            // Commit 2: Delete node
            let tx2 = manager.begin_transaction().unwrap();
            manager.delete_node_in_tx(GraphId::new(1), NodeId::new(1), tx2).unwrap();
            let commit_lsn2 = manager.commit_transaction(tx2).unwrap();

            // Verify second delta in index
            let delta2 = manager.delta_index()
                .get_node_delta(NodeId::new(1), SnapshotId::from_lsn(commit_lsn2));
            assert!(delta2.is_some());
            assert!(matches!(delta2.unwrap().record, WalRecord::DeleteNode { .. }));

            // Checkpoint at LSN between commits
            let checkpoint_lsn = (commit_lsn1 + commit_lsn2) / 2;
            manager.checkpoint(checkpoint_lsn).unwrap();

            // Verify first delta dropped (now in base), second delta remains
            let delta_after = manager.delta_index()
                .get_node_delta(NodeId::new(1), SnapshotId::from_lsn(commit_lsn2));
            assert!(delta_after.is_some());
            // Should be DeleteNode (first delta was dropped)
            assert!(matches!(delta_after.unwrap().record, WalRecord::DeleteNode { .. }));

            // Checkpoint after second commit
            manager.checkpoint(commit_lsn2 + 10).unwrap();

            // Verify all deltas dropped
            assert_eq!(manager.delta_index().node_delta_count(), 0);
        }
    }
    ```

    **Test coverage:**
    - Delta populated at commit time
    - Multiple commits tracked correctly
    - Checkpoint drops applied deltas
    - Memory bounded (deltas cleaned up after checkpoint)
