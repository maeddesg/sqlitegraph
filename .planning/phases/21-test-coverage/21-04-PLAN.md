    ```rust
            enable_multilayer: true,
            multilayer_deterministic_seed: Some(42),
            multilayer_level_distribution_base: Some(16),
            // ... other config
        };

        let mut hnsw = HnswIndex::new("test_layers", config).unwrap();

        // Insert 100 vectors
        for i in 0..100 {
            let vector = vec![i as f32; 64];
            hnsw.insert_vector(&vector, None).unwrap();
        }

        // Verify nodes are distributed across layers
        let stats = hnsw.statistics().unwrap();
        // Layer 0 should have most nodes (~94)
        // Layer 1 should have some nodes (~6)
        // Layer 2 should have very few (~0-1)
        assert!(stats.layer_stats[0].0 > 90, "Layer 0 should have most nodes");
        assert!(stats.layer_stats[1].0 > 0, "Layer 1 should have some nodes");

        // Verify higher layers have fewer nodes
        assert!(stats.layer_stats[0].0 > stats.layer_stats[1].0,
                "Layer 0 > Layer 1");
    }
    ```

    This verifies the MultiLayerNodeManager correctly tracks which nodes are in which layers.
