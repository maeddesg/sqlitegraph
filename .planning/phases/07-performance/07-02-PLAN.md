3. Consider `parking_lot::RwLock` for better performance than `std::sync::RwLock`

**Constraints:**
- Must maintain correctness of WAL recovery
- Must preserve transaction atomicity
- Cannot break existing Thread-safe behavior

use std::sync::atomic::{AtomicU64, Ordering};

pub struct ReplayStatistics {
    pub transactions_replayed: AtomicU64,
    pub operations_replayed: AtomicU64,
    pub nodes_created: AtomicU64,
    pub edges_created: AtomicU64,
    pub strings_interned: AtomicU64,
    pub nodes_deleted: AtomicU64,
    pub edges_deleted: AtomicU64,
}

impl ReplayStatistics {
    pub fn new() -> Self {
        Self {
            transactions_replayed: AtomicU64::new(0),
            operations_replayed: AtomicU64::new(0),
            nodes_created: AtomicU64::new(0),
            edges_created: AtomicU64::new(0),
            strings_interned: AtomicU64::new(0),
            nodes_deleted: AtomicU64::new(0),
            edges_deleted: AtomicU64::new(0),
        }
    }

    pub fn increment_transactions(&self) {
        self.transactions_replayed.fetch_add(1, Ordering::Relaxed);
    }

    pub fn increment_operations(&self) {
        self.operations_replayed.fetch_add(1, Ordering::Relaxed);
    }

    pub fn add_nodes(&self, count: u64) {
        self.nodes_created.fetch_add(count, Ordering::Relaxed);
    }

    pub fn snapshot(&self) -> StatisticsSnapshot {
        StatisticsSnapshot {
            transactions_replayed: self.transactions_replayed.load(Ordering::Relaxed),
            operations_replayed: self.operations_replayed.load(Ordering::Relaxed),
            nodes_created: self.nodes_created.load(Ordering::Relaxed),
            edges_created: self.edges_created.load(Ordering::Relaxed),
            strings_interned: self.strings_interned.load(Ordering::Relaxed),
            nodes_deleted: self.nodes_deleted.load(Ordering::Relaxed),
            edges_deleted: self.edges_deleted.load(Ordering::Relaxed),
        }
    }
}

#[derive(Clone, Debug)]
pub struct StatisticsSnapshot {
    pub transactions_replayed: u64,
    pub operations_replayed: u64,
    pub nodes_created: u64,
    pub edges_created: u64,
    pub strings_interned: u64,
    pub nodes_deleted: u64,
    pub edges_deleted: u64,
}
```

**Update replayer/mod.rs:**
- Change `Arc<Mutex<ReplayStatistics>>` to `Arc<ReplayStatistics>`
- Remove `.lock().unwrap()` calls when accessing statistics
- Use `statistics.increment_*()` methods instead
- Use `statistics.snapshot()` for final results

**Key Implementation Notes:**
- Use `Ordering::Relaxed` for counters (no synchronization needed beyond atomicity)
- Use `Ordering::Acquire/Release` only if needed for synchronization
- `snapshot()` creates consistent point-in-time view
- No mutex contention on statistics updates

DO NOT: Use `SeqCst` unless necessary for correctness (Relaxed is faster)

//! Lock contention benchmarks for WAL recovery.
//!
//! Measures performance under high parallelism to identify
//! lock contention points and validate optimizations.

use std::time::Duration;
use std::sync::{Arc, Mutex};
use criterion::{Criterion, criterion_group, criterion_main};
use sqlitegraph::{GraphConfig, open_graph};

mod bench_utils;
use bench_utils::create_benchmark_temp_dir;

const WARM_UP_TIME: Duration = Duration::from_secs(3);
const MEASURE_TIME: Duration = Duration::from_secs(15);

/// Benchmark lock contention with high parallelism
fn bench_recovery_lock_contention(criterion: &mut Criterion) {
    let mut group = criterion.benchmark_group("lock_contention");
    group.warm_up_time(WARM_UP_TIME);
    group.measurement_time(MEASURE_TIME);

    for parallelism in [1, 2, 4, 8].iter() {
        group.bench_with_input(
            criterion::BenchmarkId::new("parallel_recovery", parallelism),
            parallelism,
            |b, &parallel_threads| {
                b.iter_batched(|| {
                    let temp_dir = create_benchmark_temp_dir();
                    let db_path = temp_dir.path().join("benchmark.db");

                    // Create graph
                    let mut config = GraphConfig::native();
                    config.max_parallel_transactions = parallel_threads;

                    let graph = open_graph(&db_path, &config)
                        .expect("Failed to create graph");

                    // Generate WAL with 100 transactions
                    for i in 0..100 {
                        let _ = graph.insert_node(crate::NodeSpec {
                            kind: "Node".to_string(),
                            name: format!("node_{}", i),
                            file_path: None,
                            data: serde_json::json!({"id": i}),
                        });
                    }

                    // Force WAL flush
                    drop(graph);

                    // Measure recovery with lock contention
                    let start = std::time::Instant::now();
                    let _graph = open_graph(&db_path, &config)
                        .expect("Failed to recover graph");
                    let duration = start.elapsed();

                    // Cleanup
                    drop(_graph);
                    drop(temp_dir);

                    duration
                }, |batch| {
                    // Batch setup
                });
            },
        );
    }

    group.finish();
}

/// Benchmark snapshot contention
fn bench_snapshot_contention(criterion: &mut Criterion) {
    let mut group = criterion.benchmark_group("snapshot_contention");
    group.warm_up_time(WARM_UP_TIME);
    group.measurement_time(MEASURE_TIME);

    for thread_count in [1, 4, 16].iter() {
        group.bench_with_input(
            criterion::BenchmarkId::new("concurrent_snapshots", thread_count),
            thread_count,
            |b, &threads| {
                b.iter_batched(|| {
                    let temp_dir = create_benchmark_temp_dir();
                    let db_path = temp_dir.path().join("benchmark.db");

                    let graph = Arc::new(Mutex::new(
                        open_graph(&db_path, &GraphConfig::native())
                            .expect("Failed to create graph")
                    ));

                    // Warm up graph with data
                    {
                        let g = graph.lock().unwrap();
                        for i in 0..100 {
                            let _ = g.insert_node(crate::NodeSpec {
                                kind: "Node".to_string(),
                                name: format!("node_{}", i),
                                file_path: None,
                                data: serde_json::json!({"id": i}),
                            });
                        }
                    }

                    // Measure concurrent snapshot acquisition
                    let start = std::time::Instant::now();

                    // Spawn threads that acquire snapshots
                    let handles: Vec<_> = (0..*threads)
                        .map(|_| {
                            let graph_clone = graph.clone();
                            std::thread::spawn(move || {
                                let g = graph_clone.lock().unwrap();
                                let _ = g.acquire_snapshot();
                            })
                        })
                        .collect();

                    // Wait for all threads
                    for handle in handles {
                        handle.join().unwrap();
                    }

                    let duration = start.elapsed();

                    // Cleanup
                    drop(graph);
                    drop(temp_dir);

                    duration
                }, |batch| {
                    // Batch setup
                });
            },
        );
    }

    group.finish();
}

criterion_group!(benches, bench_recovery_lock_contention, bench_snapshot_contention);
criterion_main!(benches);
```
